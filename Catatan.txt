cara gcp CLOUD RUN :
- buka google cloud console
cd namaproject_ada_file_app.py_sama_Dockerfile
- lalu masukan file app.py Dockerfile requirements.txt sama Model.h5

buka Google Cloud Console

docker build -t wayang-predict-package .

docker tag wayang-predict-package gcr.io/semar-cloud-section/wayang-predict-package

docker push gcr.io/semar-cloud-section/wayang-predict-package

gcloud run deploy wayang-prediction \
    --image gcr.io/semar-cloud-section/wayang-predict-package\
    --platform managed \
    --memory=4Gi \
    --allow-unauthenticated
